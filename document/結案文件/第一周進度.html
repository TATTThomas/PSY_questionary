共同行為

本地資料使用網頁的localstorage
  1. 檢測本地資料是否存在, 若存在先寄API至server回傳result:success才繼續
  2. 檢測系統解析度要求或強制調整為1024*768,建立一個覆蓋整個網頁的canvas
  3. 彈出起始彈窗
  4. 進入練習或實作模式
  5. 練習模式結束reload本頁(也可直接重至,再重新跳出教學彈窗)
  6. 實作模式結束儲存結果於本地,寄出“./GQ/SQ/saveData”,若回傳result:success,刪除本地資料,利用“./GQ/EQ”跳轉回選單頁
  7. 練習模式不紀錄

function buildJson(num,names,elements){
    var json = {} 
    json[Trail] = num + 1
    for(var i=0;i<names.length;i++)
        json[names[i]] = elements[i]
    return json
}
function buildJsonList(str,names){
    jsons = str.spilt('~')
    output = []
    for(var i in jsons)
        var json = buildJson(i,names,jsons[i].spilt('_') 
        output.push(json)
    return output
}
function buildClassJson(str,names){
    lists = str.spilt('-')
    output = {} 
    for(var i in lists)
        output[i+1] = buildJsonList(lists[i],names)
    return output
}
function SaveOneInDB(type,str){
    switch type:
        case A:
            return buildJsonList(str,namesList_one.A)
        case B:
            return buildJsonList(str,namesList_one.B)
        case C:
            return buildClassJson(str,namesList_one.C)
        case G:
            return buildClassJson(str,namesList_one.G)
        case H:
            return buildClassJson(str,namesList_one.H)
        case I:
            return buildJsonList(str,namesList_one.I)
        case K:
            return buildJsonList(str,namesList_one.K)
}

function buildGroupJson(str,names)
{
    var elements = str.spilt('_')
    var json = {} 
    for(var i=0;i<names.length;i++)
        json[names[i]] = elements[i]
    return json
}
function SaveGroupInDB(type,str){
    return buildGroupJson(str,namesList_group[type])
}
